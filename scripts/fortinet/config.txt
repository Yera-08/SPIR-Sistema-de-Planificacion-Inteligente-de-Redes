config system global
    set hostname "SPIR-Fortinet-FW"
    set admin-sport 80
    set admin-https-port 443
end
config system interface
    edit "port1"
        set vdom "root"
        set ip 192.168.100.3 255.255.255.0
        set allowaccess ping https ssh http fgfm capwap
        set description "Uplink to ISP"
    next
    edit "port2"
        set vdom "root"
        set ip 192.168.1.3 255.255.255.0
        set allowaccess ping https ssh http fgfm capwap
        set description "Management Interface"
    next
end
config router static
    edit 1
        set dst 0.0.0.0 0.0.0.0
        set gateway 192.168.100.254
        set device "port1"
    next
end
config system ntp
    set ntpsync enable
    set type custom
    config ntpserver
        edit 1
            set server "1.pe.pool.ntp.org"
        next
    end
end
config system timezone
    set timezone 34
end
